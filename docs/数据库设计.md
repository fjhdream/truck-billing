# 建表sql

``` sql
CREATE TABLE "user" (
    id VARCHAR(128) PRIMARY KEY,
	user_name VARCHAR(128) NOT NULL,
	avatar_url text
);

CREATE TYPE role_type AS ENUM ('OWNER', 'ADMIN', 'DRIVER');
CREATE TABLE role (
	id uuid PRIMARY KEY,
	user_id VARCHAR(128) NOT NULL REFERENCES "user" (id),
	type role_type NOT NULL,
	UNIQUE (user_id, "type")
);

CREATE TABLE team (
	id uuid PRIMARY KEY,
	team_name VARCHAR(128) NOT NULL,
	user_id VARCHAR(128) NOT NULL REFERENCES "user"(user_id)
);

CREATE TABLE car (
	id uuid PRIMARY KEY,
	car_plate_number VARCHAR(64) NOT NULL,
	team_id uuid REFERENCES team(id)
);

CREATE TABLE team_car (
	id uuid PRIMARY KEY,
	car_id uuid REFERENCES car(id),
	team_id uuid REFERENCES team(id)
);

CREATE TABLE team_driver (
	id uuid PRIMARY KEY,
	user_id VARCHAR(128) REFERENCES "user"(id),
	team_id uuid REFERENCES team(id)
);

CREATE TYPE item_type as ENUM('BASIC', 'COUSTOM');
CREATE TABLE item (
	id uuid PRIMARY KEY,
	type item_type NOT NULL,
	name VARCHAR(128) NOT NULL,
	team_id uuid REFERENCES team(id),
	icon_url text 
);

CREATE TABLE billing (
	id uuid PRIMARY KEY,
	name VARCHAR(128) NOT NULL,
	team_id uuid REFERENCES team(id),
	start_time TIMESTAMP,
	end_time TIMESTAMP
);

CREATE TABLE billing_item (
	id uuid PRIMARY KEY,
	billing_id uuid REFERENCES billing(id),
	cost money CHECK (cost > 0 :: money) NOT NULL,
	item_id uuid REFERENCES item(id),
	time TIMESTAMP NOT NULL
);

```
